async function processInvoice(invoiceId) {
  const invoice = await invoiceService.fetch(invoiceId);
  invoice.validate();
  invoice.applyDiscount(0.15);
  const result = await invoiceService.send(invoice);

  return {
    status: result.status,
    sentAt: result.sentAt,
  };
}
